# Algorithm

### 일곱 난쟁이

> 문제 링크 : https://www.acmicpc.net/problem/2309



* 조합을 사용하여 9명중 7명을 뽑아서 키의 합이 100이 되는 배열을 찾음




###### 풀이

~~~java
import java.util.Arrays;
import java.util.Scanner;

/*
 * 2309. 일곱 난쟁이
 * JAVA_서울_6반_엄재웅
 */

/*
 * TODO
 * - 조합으로 7명을 뽑아서 합이 100이 되면 출력
 */

public class Solution {
	private static int[] dwarfs;
	private static int[] selected;
	private static boolean isComplete;
	
	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		StringBuilder sb = new StringBuilder();
		
		dwarfs = new int[9];				// 입력받을 난쟁이 배열 생성
		selected = new int[7];
		
		for (int i = 0; i < 9; i++) 			// 난쟁이 키 입력받기
			dwarfs[i] = sc.nextInt();
		
		Comb(0, 0);								// 조건을 만족하는 난쟁이들 찾기
		
		// 난쟁이들 키 오름차순 정렬
		Arrays.sort(selected);
		
		// StringBuilder에 담기
		for (int i = 0; i < selected.length; i++) {
			sb.append(selected[i]).append('\n');
		}

		System.out.println(sb);
	}

	private static void Comb(int cnt, int start) {
		if(cnt == 7) {
			// 완료
			int sum = 0;
			
			// 키의 합 구하기
			for (int i = 0; i < 7; i++) 
				sum += selected[i];
			
			// 키가 100이면 플래그
			if(sum == 100)
				isComplete = true;
			
			return;
		}
		
		// 조합 만들기
		for (int i = start; i < 9; i++) {
			// 정답을 찾으면 그냥 다 탈출
			if(isComplete)	return;
			selected[cnt] = dwarfs[i];
			Comb(cnt + 1, i + 1);
		}
	}
}

~~~
