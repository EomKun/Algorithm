# Algorithm

### 탑

> 문제 링크 : https://www.acmicpc.net/problem/2493



###### 풀이

~~~java
import java.io.BufferedReader;
import java.io.InputStreamReader;
import java.util.Stack;

/* TODO
 * 1. 탑 개수(N) 입력 받기
 * 2. 검사를 위한 Stack 생성
 * 3. 탑의 높이를 하나씩 입력 받으면서 검사
 *    1. stack이 안 비었을 때
 *      1. input <= top 일 때
 *        1. 해당 Idx 출력 후
 *        2. input값과 input idx를 push후 반복문 탈출
 *      2. input > top 일 때
 *        1. top pop
 *    2. stack이 비었을 때
 *      1.0을 출력하고 현재 입력값과 idx push
 */

/*
 * 단순 이중 for문으로는 시간초과
 * Stack을 이용하여 해결하여야 한다
 */

class bj2493_Tower_엄재웅 {
	public static void main(String[] args) throws Exception{
		BufferedReader br = new BufferedReader(new InputStreamReader(System.in)); 
		StringBuilder sb = new StringBuilder();
		int N = Integer.parseInt(br.readLine());
		String[] towers = br.readLine().split(" ");
		Stack<int[]> examine = new Stack<>();
		
		for(int n = 1; n <= N; n++) {
			int t = Integer.parseInt(towers[n-1]);
			while(!examine.isEmpty()) {
				if(t <= examine.peek()[1]) {
					sb.append(examine.peek()[0] + " ");
					examine.push(new int[] {n, t});
					break;
				} else
					examine.pop();
			}
			
			if(examine.isEmpty()) {
				sb.append("0 ");
				examine.push(new int[] {n, t});
			}
		}
		
		System.out.println(sb);
	}
}
~~~
